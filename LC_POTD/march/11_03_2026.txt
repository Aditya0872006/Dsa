class Solution {
public:
    int bitwiseComplement(int n) 
    {
        if(n==0) return 1;
        int ans = 0;
        int i=1;
        while(n)
        {
            int c=n&1;
            if(c == 0) ans+=i;
            
            i*=2;
            n>>=1;
        }
        return ans;
        
    }
};
