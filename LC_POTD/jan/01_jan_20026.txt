class Solution {
public:
    vector<int> plusOne(vector<int>& digits) 
    {
        int last = digits.size()-1;
        if(digits.back() < 9) 
        {
            digits.back()+=1;
            return digits;
        }
        while(last>=0 && digits[last] == 9)
        {
            digits[last]=0;
            last--;
        }
        if(last==-1)
        {
            digits.insert(digits.begin(),1);
        }
        else digits[last]++;
        return digits;
        
        
    }
};
