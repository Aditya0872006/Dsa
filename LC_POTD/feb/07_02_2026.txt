class Solution {
public:
    int minimumDeletions(string s) 
    {
        stack<char> st;
        int n=s.size();
        int c=0;
        for(int i=n-1; i>=0; i--)
        {
            if(!st.empty() && (st.top() == 'a' && s[i]=='b'))
            {
                st.pop();
                c++;
            }
            else st.push(s[i]);
        }
        return c;
    }
};
