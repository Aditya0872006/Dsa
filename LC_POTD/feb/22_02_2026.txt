class Solution {
public:
    int binaryGap(int num) 
    {
        string s = bitset<32>(num).to_string();
        int ans=0;

        int i=0,n=s.size();
        int last=-1;
        while(i<n)
        {
            if(s[i]=='1')
            {
                if(last == -1) last=i;
                else 
                {
                    ans = max(ans,i-last);
                    last=i;
                }
            }
            i++;
        }
        return ans;
        
    }
};
